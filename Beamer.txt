% Key Findings Section
\section{Key Findings}
\begin{frame}{Notable Observations}
    \textbf{1. Pattern Discovery}
    \begin{itemize}
        \item Hotspot emergence in densely populated regions
        \item Strong correlation between mobility and spread velocity
    \end{itemize}
    \vspace{0.5em}
    \textbf{2. Effectiveness of Interventions}
    \begin{itemize}
        \item Early lockdowns slowed down the infection curve
        \item Social distancing significantly reduced contact rate
    \end{itemize}
    \vspace{0.5em}
    \textbf{3. Critical Risk Factors}
    \begin{itemize}
        \item Inequitable healthcare resource allocation
        \item Influx of individuals from high-risk zones
    \end{itemize}
\end{frame}
